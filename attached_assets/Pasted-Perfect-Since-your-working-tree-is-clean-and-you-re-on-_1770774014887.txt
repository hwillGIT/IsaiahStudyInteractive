+import { useChapterData } from '../hooks/useChapterData';
+import { ChapterNavigation } from './ChapterNavigation';
+
+export function ChapterTemplate({ chapterNumber, maxChapter = 66 }: { chapterNumber: number; maxChapter?: number }) {
+  const { data, loading, error } = useChapterData(chapterNumber);
+  if (loading) return <div>Loading...</div>;
+  if (error || !data) return <div>Error loading chapter</div>;
+  return (
+    <div>
+      <ChapterNavigation currentChapter={chapterNumber} maxChapter={maxChapter} basePath="/chapter" />
+      <h1>{data.title}</h1>
+      <p>{data.subtitle}</p>
+    </div>
+  );
+}
